Em um pseudo codigo, imagino algo assim:


f_print(int X)
{
       printf ("%d", X)
       if (X == 0) {
              printf("\n")
              return
       }
       else {
            printf(" - ");
            f_print(X - 1)
       }
}

main
{
         fprintf("Print: ");
         f_print (atoi(argv[1]));
}
